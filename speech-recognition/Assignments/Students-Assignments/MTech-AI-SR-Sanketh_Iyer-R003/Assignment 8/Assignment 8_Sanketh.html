def data_preprocessing(df):
    df["Messages"] = df["Messages"].apply(lambda x: re.sub('[^a-zA-Z]',' ',x))
    df["Messages"] = df["Messages"].apply(lambda x: x.lower())
    df["Messages"] = df["Messages"].apply(lambda x: x.replace("[^A-Za-z]",""))
    df["Messages"] = df["Messages"].apply(lambda x: x.replace("'",""))
    df["Messages"] = df["Messages"].apply(lambda x: x.replace("\s+"," "))
    df["Messages"] = df["Messages"].apply(lambda x: " ".join([word for word in x.split() if word not in punct]))
    df["Messages"] = df["Messages"].apply(lambda x: " ".join([word for word in x.split() if word not in stopwords and len(word)>1]))
    df["Messages"] = "start "+df["Messages"]+" end"
    return df

def create_mfcc_features(path):
    mfccs = []
    try:
        x , sr = librosa.load(path, res_type='kaiser_fast')
        mfccs = librosa.feature.mfcc(y=x, sr=sr, n_mfcc=128)
        mfccs = np.mean(mfccs.T,axis=0)
    except:
        print('Error reading audio')
    return mfccs
